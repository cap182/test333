Canciones antes de 1990

SELECT [NOMBRE CANCION], AÑO, NOMBRE FROM CANCION, ARTISTA
WHERE CANCION.ARTISTA = ARTISTA.ID AND CANCION.AÑO < 1989;

2)

Cancion Artista

SELECT [NOMBRE CANCION], NOMBRE
FROM CANCION, ARTISTA
WHERE CANCION.ARTISTA = ARTISTA.ID;

3)

Canciones en Ingles o Español

SELECT *
FROM CANCION
WHERE CANCION.IDIOMA = 2
OR CANCION.IDIOMA =1;


4)

Cancion del album dude ranch o de artista Nirvana

SELECT [NOMBRE CANCION], DURACION
FROM CANCION
WHERE CANCION.ALBUM = 'DUDE RANCH' 
OR CANCION.ARTISTA = 11;


5)

Instrumentales

SELECT [NOMBRE CANCION], INSTRUMENTAL, NOMBRE
FROM CANCION, ARTISTA
WHERE CANCION.ARTISTA = ARTISTA.ID AND CANCION.INSTRUMENTAL = TRUE;


6)

Rock o MEtal

SELECT [NOMBRE CANCION]
FROM CANCION
WHERE (CANCION.[CODIGO-GENERO] = 4 OR CANCION.[CODIGO-GENERO] =1);


7)

Rock

SELECT [NOMBRE CANCION], NOMBRE FROM CANCION, GENERO
WHERE CANCION.[CODIGO-GENERO] = GENERO.ID AND GENERO.ID = 1;


8)

Albunes especificos

SELECT [NOMBRE CANCION], NOMBRE FROM CANCION, ARTISTA
WHERE CANCION.ARTISTA = ARTISTA.ID AND CANCION.ALBUM = 'REVEAL'
UNION
SELECT [NOMBRE CANCION], NOMBRE FROM CANCION, ARTISTA
WHERE CANCION.ARTISTA = ARTISTA.ID AND CANCION.ALBUM = 'RAMONES MANIA'
UNION SELECT [NOMBRE CANCION], NOMBRE FROM CANCION, ARTISTA
WHERE CANCION.ARTISTA = ARTISTA.ID AND CANCION.ALBUM = 'INCESTICIDE';



9)

Canciones que no son de la decada de los 90s

(SELECT [NOMBRE CANCION] FROM CANCION
WHERE (CANCION.AÑO < 1990))
UNION 
(SELECT [NOMBRE CANCION] FROM CANCION
WHERE (CANCION.AÑO > 2000));


10)

Rock o Punk antes de los 90s

SELECT [NOMBRE CANCION], DURACION, AÑO, NOMBRE
FROM CANCION, GENERO
WHERE CANCION.[CODIGO-GENERO] = GENERO.ID AND GENERO.ID = 1 OR GENERO.ID = 3 AND CANCION.AÑO < 1980;
